 19 /**
 20  * @fileoverview Defines GetAccountHierarchy, a code example that illustrates
 21  * how to retrieve the account hierarchy under an account.
 22  */
 23 
 24 goog.provide('google.ads.adwords.examples.v201109.GetAccountHierarchy');
 25 
 26 goog.require('google.ads.adwords.AdWordsService.v201109');
 27 goog.require('google.ads.adwords.examples.ExampleBase');
 28 goog.require('google.ads.adwords.v201109.AdExtensionOverride');
 29 goog.require('google.ads.adwords.v201109.RequestHeader');
 30 goog.require('google.ads.adwords.v201109.ServicedAccountSelector');
 31 goog.require('google.ads.adwords.v201109.ServicedAccountService');
 32 
 33 /**
 34  * This code example illustrates how to retrieve the account hierarchy under
 35  * an account.
 36  *
 37  * Tags: ServicedAccountService.get
 38  *
 39  * @extends google.ads.adwords.examples.ExampleBase
 40  * @constructor
 41  */
 42 google.ads.adwords.examples.v201109.GetAccountHierarchy = function() {
 43   google.ads.adwords.examples.ExampleBase.call(this);
 44   this.description = 'This code example illustrates how to retrieve the ' +
 45       'account hierarchy under an account.';
 46 };
 47 goog.inherits(google.ads.adwords.examples.v201109.GetAccountHierarchy,
 48     google.ads.adwords.examples.ExampleBase);
 49 
 50 /**
 51  * Runs the code example.
 52  *
 53  * @param {google.ads.adwords.AdWordsUser} user AdWords user running the code
 54  *     example.
 55  * @param {function} callback the callback method to be called once this example
 56  *     is complete.
 57  */
 58 google.ads.adwords.examples.v201109.GetAccountHierarchy.prototype.run =
 59     function(user, callback) {
 60   // Get the ServicedAccountService.
 61   user.getConfig().setClientCustomerId(null);
 62   var servicedAccountService = user.getService(
 63       google.ads.adwords.AdWordsService.v201109.ServicedAccountService);
 64 
 65   servicedAccountService.requestHeader =
 66       new google.ads.adwords.v201109.RequestHeader();
 67 
 68   var selector = new google.ads.adwords.v201109.ServicedAccountSelector();
 69   selector.enablePaging = false;
 70 
 71   try {
 72     servicedAccountService.get(selector,
 73       goog.bind(function(graph) {
 74         if (graph && graph.accounts) {
 75           // Create map from customer id to parent and child links.
 76           var childLinks = [];
 77           var parentLinks = [];
 78           goog.array.forEach(graph.links, function(link) {
 79             if (childLinks[link.managerId.id] == null) {
 80               childLinks[link.managerId.id] = [];
 81             }
 82             childLinks[link.managerId.id].push(link);
 83             if (parentLinks[link.managerId.id] == null) {
 84               parentLinks[link.clientId.id] = [];
 85             }
 86             parentLinks[link.clientId.id].push(link);
 87           });
 88           // Create map from customer id to account and find root account.
 89           var accounts = [];
 90           var rootAccount = null;
 91           goog.array.forEach(graph.accounts, function(account) {
 92             accounts[account.customerId] = account;
 93             if (!parentLinks[account.customerId]) {
 94               rootAccount = account;
 95             }
 96           });
 97           // Display account tree.
 98           var displayTree =
 99               goog.bind(function(account, link, accounts, links, depth) {
100             var output = new Array(depth * 2).join('-') + '%s, %s';
101             if (link) {
102               output += ' (%s, %s)';
103             }
104             if (!link) {
105               this.writeOutput(output, account.login, account.customerId);
106             } else {
107               this.writeOutput(output, account.login, account.customerId,
108                   link.typeOfLink, link.descriptiveName);
109             }
110             if (links[account.customerId] != null) {
111               goog.array.forEach(links[account.customerId],
112                   function(childLink) {
113                 var childAccount = accounts[childLink.clientId.id];
114                 displayTree(childAccount, childLink, accounts, links,
115                     depth + 1);
116               });
117             }
118           }, this);
119           displayTree(rootAccount, null, accounts, childLinks, 0);
120         }
121         callback();
122       }, this),
123       goog.bind(function(soapException) {
124         this.writeOutput('Failed to get account hierarchy. Soap Fault ' +
125             'says "%s"', soapException.getInnerException().getFaultString());
126         callback();
127       }, this)
128     );
129   } catch (ex) {
130     this.writeOutput('An exception occurred while running the code example.');
131     callback();
132   }
133 };
