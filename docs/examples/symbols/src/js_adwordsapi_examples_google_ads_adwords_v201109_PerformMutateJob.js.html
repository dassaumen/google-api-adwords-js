 19 /**
 20  * @fileoverview Defines PerformMutateJob, a code example that shows how to
 21  * add keywords using the MutateJobService.
 22  */
 23 
 24 goog.provide('google.ads.adwords.examples.v201109.PerformMutateJob');
 25 
 26 goog.require('google.ads.adwords.AdWordsService.v201109');
 27 goog.require('google.ads.adwords.examples.ExampleBase');
 28 goog.require('google.ads.adwords.v201109.AdGroupAdOperation');
 29 goog.require('google.ads.adwords.v201109.AdGroupCriterionOperation');
 30 goog.require('google.ads.adwords.v201109.BiddableAdGroupCriterion');
 31 goog.require('google.ads.adwords.v201109.BulkMutateJobPolicy');
 32 goog.require('google.ads.adwords.v201109.BulkMutateJobSelector');
 33 goog.require('google.ads.adwords.v201109.Keyword');
 34 goog.require('google.ads.adwords.v201109.KeywordMatchType');
 35 goog.require('google.ads.adwords.v201109.MutateJobService');
 36 
 37 /**
 38  * This code example shows how to add ads and keywords using the
 39  * MutateJobService.
 40  *
 41  * Tags: MutateJobService.mutate
 42  *
 43  * @extends google.ads.adwords.examples.ExampleBase
 44  * @constructor
 45  */
 46 
 47 google.ads.adwords.examples.v201109.PerformMutateJob =
 48     function() {
 49   google.ads.adwords.examples.ExampleBase.call(this);
 50   this.description = 'This code example shows how to add ads and keywords ' +
 51       'using the MutateJobService.';
 52 };
 53 goog.inherits(
 54     google.ads.adwords.examples.v201109.PerformMutateJob,
 55     google.ads.adwords.examples.ExampleBase);
 56 
 57 /**
 58  * Time between every status fetch.
 59  * @type number
 60  * @const
 61  */
 62 google.ads.adwords.examples.v201109.PerformMutateJob.prototype.RETRY_INTERVAL =
 63     30000;
 64 
 65 /**
 66  * Maximum number of retries for fetching status before give up.
 67  * @type number
 68  * @const
 69  */
 70 google.ads.adwords.examples.v201109.PerformMutateJob.prototype.RETRIES_COUNT =
 71     30;
 72 
 73 /**
 74  * Number of keywords to create.
 75  * @type number
 76  * @const
 77  */
 78 google.ads.adwords.examples.v201109.PerformMutateJob.prototype.KEYWORD_NUMBER =
 79     100;
 80 
 81 /**
 82  * Runs the code example.
 83  *
 84  * @param {google.ads.adwords.AdWordsUser} user AdWords user running the code
 85  *     example.
 86  * @param {function} callback the callback method to be called once this example
 87  *     is complete.
 88  */
 89 google.ads.adwords.examples.v201109.PerformMutateJob.prototype.
 90     run = function(user, callback) {
 91   // Get the BulkMutateJobService.
 92   var mutateJobService = user.getService(
 93       google.ads.adwords.AdWordsService.v201109.MutateJobService);
 94 
 95   var adGroupId = this._T('INSERT_ADGROUP_ID_HERE');
 96 
 97   // Create AdGroupCriterionOperations to add keywords.
 98   var adGroupCriterionOperations = [];
 99 
100   for (var i = 0; i < this.KEYWORD_NUMBER; i++) {
101     var keyword = new google.ads.adwords.v201109.Keyword();
102     keyword.text = goog.string.format('mars cruise %d', i);
103     if (Math.floor(Math.random() * 11) == 0) {
104       keyword.text += '!!!';
105     }
106     keyword.matchType = google.ads.adwords.v201109.KeywordMatchType.BROAD;
107 
108     var criterion = new google.ads.adwords.v201109.BiddableAdGroupCriterion();
109     criterion.adGroupId = adGroupId;
110     criterion.criterion = keyword;
111 
112     var adGroupCriterionOperation =
113         new google.ads.adwords.v201109.AdGroupCriterionOperation();
114     adGroupCriterionOperation.operator =
115         google.ads.adwords.v201109.Operator.ADD;
116 
117     adGroupCriterionOperation.operand = criterion;
118     adGroupCriterionOperations[i] = adGroupCriterionOperation;
119   }
120 
121   // You can specify up to 3 job IDs that must successfully complete before this
122   // job can be processed.
123   var policy = new google.ads.adwords.v201109.BulkMutateJobPolicy();
124   policy.prerequisiteJobIds = [];
125 
126   var self = this;
127 
128   // Call mutate to create a new job.
129   try {
130     mutateJobService.mutate(adGroupCriterionOperations, policy,
131       goog.bind(function(bulkJob) {
132         var bulkJobId = bulkJob.id;
133         self.writeOutput('Job ID %d was created successfully.', bulkJobId);
134 
135         // Creating selector to retrieve job status and wait for it to complete.
136         var retries = 0;
137         var callbackFunction = goog.bind(function() {
138           var selector = new google.ads.adwords.v201109.BulkMutateJobSelector();
139           selector.jobIds = [bulkJobId];
140           var completed = false;
141           var job;
142 
143           mutateJobService.get(selector, goog.bind(function(allJobs) {
144             if (allJobs && allJobs.length > 0) {
145               job = allJobs[0];
146               if (job.status ==
147                   google.ads.adwords.v201109.BasicJobStatus.COMPLETED) {
148                 self.writeOutput('Job finished with status "%s".', job.status);
149                 completed = true;
150               } else if (job.status ==
151                   google.ads.adwords.v201109.BasicJobStatus.FAILED) {
152                 self.writeOutput('Job failed with reason "%s".',
153                     job.statusResponse.failureReason);
154               } else if (retries < self.RETRIES_COUNT) {
155                 self.writeOutput('Current status is "%s", waiting %d to' +
156                     'retry...', var_args);
157                 setTimeout(callbackFunction(), self.RETRY_INTERVAL);
158               }
159             }
160             if (completed) {
161               mutateJobService.getResult(selector,
162                 goog.bind(function(resultResponse) {
163                   var results = resultResponse.simpleMutateResult.results;
164                   if (results) {
165                     for (var i = 0, len = results.length; i < len; i++) {
166                       var result = results[i];
167                       if (result.placeHolder) {
168                         self.writeOutput('Operation [%d] - FAILED', i);
169                       } else {
170                         self.writeOutput('Operation [%d] - SUCCEEDED', i);
171                       }
172                     }
173                   }
174                   var errors = resultResponse.simpleMutateResult.errors;
175                   if (errors) {
176                     for (var i = 0, len = errors.length; i < len; i++) {
177                       var error = errors[i];
178                       self.writeOutput('Error, reason: "%s", trigger: "%s", ' +
179                           'field path: %s', error.reason, error.trigger,
180                           error.fieldPath);
181                     }
182                   }
183                 }, this), goog.bind(function(soapException) {
184                   this.writeOutput('Failed to get job result. Soap Fault ' +
185                       'says "%s"', soapException.getInnerException().
186                       getFaultString());
187                   callback();
188                 }, this));
189             } else {
190               this.writeOutput('Job failed to complete after %d retries',
191                   self.RETRY_COUNT);
192             }
193           }), goog.bind(function(soapException) {
194               this.writeOutput('Failed to get job status. Soap Fault says "%s"',
195                   soapException.getInnerException().getFaultString());
196               callback();
197             }, this));
198           }, this);
199 
200         setTimeout(callbackFunction(), self.RETRY_INTERVAL);
201 
202       }, this), goog.bind(function(soapException) {
203         this.writeOutput('"Failed to create job. Soap Fault says "%s"',
204             soapException.getInnerException().getFaultString());
205         callback();
206       }, this));
207   } catch (ex) {
208     this.writeOutput('An exception occurred while running the code example.');
209     callback();
210   }
211 };
